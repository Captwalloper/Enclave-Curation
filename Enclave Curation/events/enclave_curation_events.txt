    option = {

        trigger = { NOT = { event_target:enclave_11 = { is_same_value = root } } }
        root = { change_variable = { which = page_index value = 1 } }
        country_event = { id = enclave_curation.100 }

    option = {
        name = "enclave_refresh"
        custom_tooltip = "enclave_refresh_TOOLTIP"
        country_event = { id = enclave_curation.100 }
    }

    after = {
        if = { # enclave selected, so not a refresh/paging pass
            limit = { NOT = { event_target:target_enclave = { is_same_value = root } } }
            invite_enclave = yes
            root = { clear_variable = page_index }
        }
    }
