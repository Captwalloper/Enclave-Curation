can_see_invite_enclave = {
	event_target:$enclave$ = {
		NOT = { is_same_value = root } # slot populated
		has_communications = root
	}
}

can_invite_enclave = {
	event_target:$enclave$ = {
		has_communications = root
		opinion = { who = root value >= 50 }
		custom_tooltip = {
			fail_text = invited_enclave_outpost_slot_unavailable
			success_text = invited_enclave_outpost_slot_available
			count_owned_fleet = {
				limit = { 
					is_inside_border = root
					has_fleet_flag = invited_enclave_outpost
				}
				count < 1
			}
		}
	}
}

can_evict_enclave = {
	event_target:$enclave$ = {
		NOT = { is_same_value = root } # slot populated
		has_communications = root
		exists = event_target:target_enclave_system
		count_owned_fleet = {
			limit = { 
				is_inside_border = root
			}
			count > 0
		}
	}
}
