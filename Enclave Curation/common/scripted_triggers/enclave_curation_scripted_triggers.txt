can_invite_enclave = {
	event_target:$enclave$ = {
		has_communications = root
		opinion = { who = root value >= 50 }
		count_owned_fleet = {
			limit = { 
				exists = space_owner
				root = { is_country = space_owner }
			}
			count < 1
		}
	}
}

can_evict_enclave = {
	event_target:$enclave$ = {
		has_communications = root
		exists = event_target:target_enclave_system
		count_owned_fleet = {
			limit = { 
				exists = space_owner
				root = { is_country = space_owner }
			}
			count > 0
		}
	}
}
